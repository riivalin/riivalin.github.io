## 語法
```sql
create procedure 預存程序名稱
    --宣告參數
as
declare
    --宣告變數
begin
    --處理內容
end
```

## 範例

```sql
--製作統計所指定商品ID的販賣數量的預存程序
create procedure uspGetSum
	@id int,
	@sum int output
as
begin
	select  @sum = sum(數量)
	from 販賣資料
	where 商品ID = @id
end
```

## 執行方法

```sql
--執行
declare @@result int --宣告變數
exec uspGetSum 1, @result output --執行sp
select @result --顯示結果

--執行結果
--6
```
